String get floatPrefix {
  final sb = StringBuffer();
  final BigInt upper52 = _upper52;
  for (int d = 0; d < 10; d++) {
    sb.writeCharCode(
        '0'.codeUnitAt(0) + ((upper52 >> 36 - (d * 4)) & _maskF).toInt());
  }
  return sb.toString();
}
