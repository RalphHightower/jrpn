void convertHMStoH(Model m) {
  final Value hr = m.x;
  final Value min = Value.fromDouble(hr.fracOp().asDouble * 100);
  final double sec = min.fracOp().asDouble * 100;
  m.resultX = Value.fromDouble(
      hr.intOp().asDouble + (min.intOp().asDouble + sec / 60) / 60);
}
